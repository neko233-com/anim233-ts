Shader Animation WebGL

Create real-time WebGL shader animations with automatic time uniform, custom uniforms, and hot-swap shader editing.

Usage

const shader = Anim233.createShaderAnim233({
  target: document.getElementById('my-canvas'),
  fragment: `
    precision mediump float;
    uniform float time;
    uniform vec2 resolution;
    void main() {
      vec2 uv = gl_FragCoord.xy / resolution;
      gl_FragColor = vec4(uv.x, uv.y, sin(time), 1.0);
    }`,
  uniforms: {
    speed: 1.0,
    color: [1.0, 0.4, 0.6]
  },
  autoplay: true
});

Custom Uniforms

const shader = Anim233.createShaderAnim233({
  target: canvas,
  fragment: `
    precision mediump float;
    uniform float time;
    uniform vec2 resolution;
    uniform float speed;
    uniform vec3 color;
    void main() {
      vec2 uv = gl_FragCoord.xy / resolution;
      float wave = sin(uv.x * 10.0 + time * speed) * 0.5 + 0.5;
      gl_FragColor = vec4(color * wave, 1.0);
    }`,
  uniforms: { speed: 2.0, color: [1.0, 0.4, 0.6] }
});

// Update at runtime
shader.setUniform('speed', 3.0);
shader.setUniform('color', [0.2, 0.8, 1.0]);
