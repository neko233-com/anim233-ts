Usage

// With tween
Anim233.tween('#box')
  .to({ x: 200 }, 500, 'easeOutBack')
  .play();

// With animate
Anim233.animate({
  targets: '#box',
  x: 200,
  ease: 'easeOutElastic'
});

// Custom easing function
Anim233.registerEasing('myEase', (t) => t * t);
Anim233.tween('#box').to({ x: 200 }, 500, 'myEase').play();

// Inline function
Anim233.tween('#box').to({ x: 200 }, 500, (t) => t * t).play();
