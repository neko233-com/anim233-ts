API Reference

Complete reference for all Anim233 exports. Version 1.0.0.

Import

import Anim233 from 'anim233-ts';
// or
import { tween, animate, timeline, stagger, ... } from 'anim233-ts';
// or via CDN global
const { tween, animate, timeline } = window.Anim233;

Anim233 Namespace (Default Export)

The default export is a singleton object containing all API methods.

Types

EasingFunction

type EasingFunction = (t: number) => number;
// Input: progress [0..1], Output: eased value [0..1]

EasingName

type EasingName = 'linear' | 'ease' | 'easeIn' | 'easeOut' | 'easeInOut'
  | 'easeInQuad' | 'easeOutQuad' | 'easeInOutQuad'
  | 'easeInCubic' | 'easeOutCubic' | 'easeInOutCubic'
  | 'easeInQuart' | 'easeOutQuart' | 'easeInOutQuart'
  | 'easeInQuint' | 'easeOutQuint' | 'easeInOutQuint'
  | 'easeInSine' | 'easeOutSine' | 'easeInOutSine'
  | 'easeInExpo' | 'easeOutExpo' | 'easeInOutExpo'
  | 'easeInCirc' | 'easeOutCirc' | 'easeInOutCirc'
  | 'easeInBack' | 'easeOutBack' | 'easeInOutBack'
  | 'easeInElastic' | 'easeOutElastic' | 'easeInOutElastic'
  | 'easeInBounce' | 'easeOutBounce' | 'easeInOutBounce'
  | string; // custom easing names

TweenValue

type TweenValue = number | string | ((el: HTMLElement, i: number, l: number) => number | string);

AnimationTarget

type AnimationTarget = string | HTMLElement | HTMLElement[] | NodeListOf<HTMLElement>;

AnimationOptions

interface AnimationOptions {
  targets?: AnimationTarget;
  properties?: Record<string, any>;
  delay?: number | ((el: HTMLElement, i: number, l: number) => number);
  duration?: number;
  ease?: EasingName | EasingFunction;
  loop?: number | boolean;
  loopDelay?: number;
  alternate?: boolean;
  reversed?: boolean;
  autoplay?: boolean;
  playbackRate?: number;
  // Callbacks
  onBegin?: (anim: AnimationInstance) => void;
  onComplete?: (anim: AnimationInstance) => void;
  onBeforeUpdate?: (anim: AnimationInstance) => void;
  onUpdate?: (anim: AnimationInstance) => void;
  onRender?: (anim: AnimationInstance) => void;
  onLoop?: (anim: AnimationInstance) => void;
  onPause?: (anim: AnimationInstance) => void;
  [key: string]: any; // CSS properties to animate
}

interface AnimationInstance {
  // Methods

  pause(): void;
  restart(): void;
  reverse(): void;
  alternate(): void;
  resume(): void;
  complete(): void;
  cancel(): void;
  revert(): void;
  reset(): void;
  seek(time: number): void;
  stretch(newDuration: number): void;
  then(callback?: () => void): Promise<void>;

  // Properties
  finished: Promise<void>;
  currentTime: number;
  progress: number;
  duration: number;
  paused: boolean;

  began: boolean;

TweenInstance

interface TweenInstance {
  // Chain methods
  to(props: Record<string, any>, duration?: number, ease?: string): TweenInstance;
  from(props: Record<string, any>, duration?: number, ease?: string): TweenInstance;
  fromTo(from: Record<string, any>, to: Record<string, any>, duration?: number, ease?: string): TweenInstance;
  set(props: Record<string, any>): TweenInstance;
  delay(ms: number): TweenInstance;
  loop(count?: number): TweenInstance;
  yoyo(enable?: boolean): TweenInstance;

  // Playback
  play(): void;
  pause(): void;
  resume(): void;
  restart(): void;
  reverse(): void;
  kill(): void;

  // Callbacks
  onBegin(cb: () => void): TweenInstance;
  onComplete(cb: () => void): TweenInstance;
  onUpdate(cb: (progress: number) => void): TweenInstance;
  onLoop(cb: () => void): TweenInstance;

  // Helpers
  append(cb: () => void): TweenInstance;
  appendInterval(ms: number): TweenInstance;
  join(): TweenInstance;

  // Properties
  duration: number;
  currentTime: number;
  progress: number;
  isPlaying: boolean;
  isPaused: boolean;
  isComplete: boolean;
}

interface StaggerOptions {
  start?: number;

  reversed?: boolean;

}

Global Registration

When loaded via CDN or UMD, Anim233 registers on window:

window.Anim233;  // The singleton
window.anim233;  // Lowercase alias
